##- Detrend and test for similar distributions

print('Mann-Whitney-Wilcoxon (MWW) RankSum Test')
z_stat, p_val = stats.ranksums(data1.values-mean(data1.values), data2.values-mean(data2.values))
print("MWW RankSum p-value for periods 1 and 2 = {:f}".format(p_val))
print('(If P <= 0.05, we are highly confident that the distributions significantly differ.)\n')

print('Two-Sample Kolmogorov-Smirnov (KS) Test')
z_stat, p_val = stats.ks_2samp(data1.values-mean(data1.values), data2.values-mean(data2.values))
print("KS p-value for periods 1 and 2 = {:f}".format(p_val))
print('(If P <= 0.05, we are highly confident that the distributions significantly differ.)\n')
